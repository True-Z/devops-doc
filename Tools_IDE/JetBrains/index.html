" Source your .vimrc
" source ~/.vimrc

====================================================================================================
" Global Config
====================================================================================================
" system setting
set mapleader=' ' " leader键
set number " 设置行号
set relativenumber " 设置相对行号
set so=5 " 上下预留5行代码
set clipboard=unnamed " 使用系统剪切板
set ttimeoutlen=100 " 默认按下Esc后,需要等待1秒才生效,设置Esc超时时间为100ms,尽快生效
set vb t_vb= " 提示静音

" search setting
set hlsearch " 高亮显示所有搜索到的内容
set incsearch " 光标立刻跳转到搜索到内容
set nowrapscan " 搜索到最后匹配的位置后再次搜索不回到第一个匹配处
set ignorecase smartcase " 搜时忽略大小写，但在有一个或以上大写字母时仍保持对大小写敏感

" indent setting
set autoindent " 自动缩进
set shiftwidth=2 " 使用 << and >> 缩进长度为2
set tabstop=2 " tab 的长度为2
set softtabstop=2 " 退格键一次可以删除2个空格
set expandtab " tab替换为空格

====================================================================================================
" 模式/映射
" map 映射；nore 不递归；un 取消；clear 清除
" i 插入；n 普通；v 可视
====================================================================================================
" 切换vim/IDE模式
noremap <C-,> :action VimPluginToggle<CR>
" 可视列模式
noremap <leader>v <C-v>
" 退出编辑模式
inoremap kj <ESC>
" Nop
noremap <SPACE> <Nop>

====================================================================================================
" 视口/移动操作
====================================================================================================
" vim action
" 下/上一编辑点 <C-i/o>

" 标签页切换
noremap <Tab> gt
noremap <S-Tab> gT

" 视口下/上滚一行
noremap <C-h> <C-y>
noremap <C-l> <C-e>

" 视口下/上滚半页
noremap <C-j> <C-d>
noremap <C-k> <C-u>

" 视口下/上滚一页
noremap <C-S-j> <C-f>
noremap <C-S-k> <C-b>

" 下/上移当前行
noremap <S-A-j> :action MoveLineDown<CR>
noremap <S-A-k> :action MoveLineUp<CR>

" 行内移动
noremap <leader>h ^
noremap <leader>l $

" 下/上一方法移动
noremap <leader>j :action MethodDown<CR>
noremap <leader>k :action MethodUp<CR>

====================================================================================================
" Global Config
====================================================================================================
" action given
noremap <leader>/ :nohlsearch<CR> " 清除搜索高亮
" noremap <C-s> :action ReformatCode<CR> 刷新

" action Goto
noremap ga :action GotoAction<CR> " 操作
noremap gf :action GotoFile<CR> " 文件
noremap gc :action GotoClass<CR> " 类
noremap gm :action GotoSymbol<CR> " 符号
noremap gd :action GotoDeclaration<CR> " 定义
noremap gi :action GotoImplementation<CR> " 实现
noremap gu :action ShowUsages<CR> " 用途
noremap gc :action GotoTest<CR> " 测试
noremap gt :action GotoTypeDeclaration<CR> " 类型
noremap gs :action GotoSuperMethod<CR> " 父类
noremap gr :action RecentFiles<CR> " 最近的文件
noremap gh :action $EditorTranslateAction<CR> " 翻译
noremap ge :action ShowErrorDescription<CR> " 错误

" action Toggle
noremap tb :action ToggleLineBreakpoint<CR> " 断点
noremap tw :action EditorToggleUseSoftWraps<CR> " 换行
noremap te :action RevealIn<CR> " 资源管理器
noremap th :action $ShowTranslationDialogAction<CR> " 翻译
noremap tu :action de.netnexus.CamelCasePlugin.ToggleCamelCase<CR> " 命名格式转换
noremap tt :action ExpandCollapseToggleAction<CR> " 代码折叠


====================================================================================================
" Mark Config
====================================================================================================
noremap mm :action ToggleBookmark<CR> " 书签
noremap ml :action ShowBookmarks<CR> " 书签列表
" 编辑器内跳转
noremap mj :action GotoNextBookmarkInEditor<CR>
noremap mk :action GotoPreviousBookmarkInEditor<CR>
" 全局跳转
noremap mJ :action GotoNextBookmark<CR> " 下一书签
noremap mK :action GotoPreviousBookmark<CR> " 上一书签

nmap <leader>F <Plug>(easymotion-F)
